//membuat db baru
CREATE DATABASE pegawaisib5;

//menambah db
use pegawaisib5;
// proses DDL CREATE
//membuat table
CREATE TABLE jenis (
    id int PRIMARY KEY AUTO_INCREMENT,
    nama varchar(30) NOT NULL unique
);
/*id adalah nama kolom atau header kolom 
int adalah type DATABASE
PRIMARY KEY adalah key atau kunci sebuah table
AUTO_INCREMENT adalah key untuk otomatisasi penomoran
NOT NULL menandakan kolom tersebut tidak boleh kosong jika data mulai di isi
unique adalah key untuk kolom agar data tidak boleh sama*/

//untuk melihat detail table
DESC jenis;

//membuat table relasi (produk ke jenis)
CREATE TABLE produk (
    id int PRIMARY KEY AUTO_INCREMENT,
    nama varchar(30) NOT NULL UNIQUE,
    harga double NOT NULL,
    stok int NOT NULL,
    idjenis int NOT NULL REFERENCES jenis(id),
    foto varchar(30));
